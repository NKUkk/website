变形题1

不规定读取文件的名称，可以选择性的读取文件

代码


#include
#include
#include
#include
using namespace std;
int main(){
    string filename;
    char line[100];
    ifstream fp;
    cin >> filename;
    fp.open(filename);
    if (fp){
        while (fp.getline(line, 10000)){
            cout << line << endl;
            Sleep(1000);
        }
    }
    else
        cout << "no such file." << endl;
    return 0;
}
